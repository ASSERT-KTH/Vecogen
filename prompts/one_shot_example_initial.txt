/*
    Check if an integer x is within the inclusive range [low, high] and divisible by div.
    
    input:
        Four integers x, low, high, and div. 
        x is the value to check. 
        low and high define the inclusive bounds of the range. 
        div is the divisor to check divisibility.

    output:
        Returns 1 if x is within [low, high] and divisible by div; otherwise returns 0.
*/
---END_NATURAL_LANGUAGE---
/*@
    requires div != 0;
    ensures \result == (x >= low && x <= high && x % div == 0);
*/
---END_FORMAL_SPECIFICATION---

int checkConstraints(int x, int low, int high, int div)
{
    return (x >= low && x <= high && x % div == 0);
}
